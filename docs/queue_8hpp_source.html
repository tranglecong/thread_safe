338 template<typename T>
339 bool Queue<T>::waitToPush(const uint32_t timeout_ms)
340 {
341  if (!m_open_push.load(std::memory_order_acquire))
342  {
343  return false;
344  }
345 
346  auto closed_or_not_full_pred = [&]() -> bool
347  {
348  if (!m_open_push.load(std::memory_order_acquire) || m_status.load(std::memory_order_acquire) != Status::FULL)
349  {
350  return true;
351  }
352  return false;
353  };
354 
355  if (m_status.load(std::memory_order_acquire) == Status::FULL && m_settings.discard == Discard::NO_DISCARD)
356  {
357  Wait::Status result{m_wait.waitFor(std::chrono::milliseconds(timeout_ms), closed_or_not_full_pred)};
358  if (result != Wait::Status::SUCCESS || !m_open_push.load(std::memory_order_acquire))
359  {
360  return false;
361  }
362  }
363  return true;
364 }

411 template<typename T>
412 void Queue<T>::updateStatus()
413 {
414  constexpr std::size_t NO_ELEMENT{0};
415  m_size.store(m_queue.size(), std::memory_order_release);
416  if (m_size.load(std::memory_order_acquire) <= NO_ELEMENT)
417  {
418  m_status.store(Status::EMPTY, std::memory_order_release);
419  }
420  else if (m_size.load(std::memory_order_acquire) >= m_settings.size)
421  {
422  m_status.store(Status::FULL, std::memory_order_release);
